\documentclass{article}
\usepackage{xparse}
\usepackage{expl3}

\ExplSyntaxOn
% Define a new command to split a string by underscores and display each part on a new line
\NewDocumentCommand{\splitstring}{m}
{
  % Split the string by underscores and store the parts in a sequence
  \seq_set_split:Nnn \l_tmpa_seq { _ } { #1 }
  % Map each part of the sequence to display it followed by a newline
  \seq_map_inline:Nn \l_tmpa_seq { \tl_use:N ##1 \\ }
}
\ExplSyntaxOff

\begin{document}

% Example usage
\splitstring{a_b_c_d}

\end{document}
